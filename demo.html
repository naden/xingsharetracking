Xing Share Button jQuery-Plugin Demo

Blogpost zum Plugin

jQuery einbinden

<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.7.2/jquery.min.js"></script>

Plugin einbinden

<script src="jquery.xingsharetracking.min.js"></script>

Plugin aufrufen

$(function() {
 $.xingsharetracking({
  delay: 1500, 
  callback: function(url) {
   $('#message')
   .html('Danke, dass Sie den folgenden Link auf Xing geteilt haben:' + url)
   .show();
  } 
 });
});
